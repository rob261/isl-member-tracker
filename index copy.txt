<script>
/* ── Config & State ── */
let settings = { clientId: '1038199509852-let2a3mhpl464lqs4use2dij9v39qt6t.apps.googleusercontent.com', sheetId: '1D4RmX_Ph2-aNmzkrKwmuKluH4KmSifCq7x5FH-Xp0g8', ...JSON.parse(localStorage.getItem('isl_settings') || '{}') };
let members = JSON.parse(localStorage.getItem('isl_tracker') || '[]');
let accessToken = null;
let tokenClient = null;

function toggleAddModal(show) {
    document.getElementById('addMemberModal').classList.toggle('hidden', !show);
}

function openSettings() { document.getElementById('modalOverlay').classList.remove('hidden'); }
function saveSettings() {
    settings.clientId = document.getElementById('inputClientId').value;
    settings.sheetId = document.getElementById('inputSheetId').value;
    localStorage.setItem('isl_settings', JSON.stringify(settings));
    document.getElementById('modalOverlay').classList.add('hidden');
}

/* ── Logic (Simplified for brevity, re-insert full original parsing here) ── */
function onGISLoad() {
    tokenClient = google.accounts.oauth2.initTokenClient({
        client_id: settings.clientId,
        scope: 'https://www.googleapis.com/auth/spreadsheets',
        callback: (resp) => { accessToken = resp.access_token; setSyncStatus('idle'); }
    });
}

function setSyncStatus(state) {
    const dot = document.getElementById('syncDot');
    const lbl = document.getElementById('syncLabel');
    dot.className = 'sync-dot ' + state;
    lbl.textContent = state === 'idle' ? 'Synced' : 'Offline';
}

function renderTable() {
    const q = document.getElementById('searchInput').value.toLowerCase();
    const tbody = document.getElementById('tableBody');
    const filtered = members.filter(m => m.name.toLowerCase().includes(q) || (m.email && m.email.toLowerCase().includes(q)));
    
    tbody.innerHTML = filtered.map((m, i) => `
        <tr>
            <td>${i+1}</td>
            <td><strong>${m.name}</strong></td>
            <td>${m.date}</td>
            <td>${m.islMember || '—'}</td>
            <td>${m.email || '—'}</td>
            <td><button onclick="deleteMember(${m.id})" style="color:red; background:none; border:none; cursor:pointer;">Remove</button></td>
        </tr>
    `).join('');
    updateStats();
}

function updateStats() {
    const stats = [
        { label: 'Total Members', val: members.length },
        { label: 'Added Today', val: members.filter(m => m.date === new Date().toLocaleDateString()).length },
        { label: 'Automation Ready', val: members.filter(m => m.status === 'Added to automation').length },
        { label: 'ISL Members', val: members.filter(m => m.islMember?.toLowerCase().includes('yes')).length },
        { label: 'Leads', val: members.length }
    ];
    document.getElementById('statsGrid').innerHTML = stats.map(s => `
        <div class="stat-card"><div class="num">${s.val}</div><div class="lbl">${s.label}</div></div>
    `).join('');
}

function addMember() {
    const name = document.getElementById('memberName').value;
    const date = new Date().toLocaleDateString('en-US', {month:'short', day:'numeric', year:'numeric'});
    const m = { id: Date.now(), name, date, email: 'Pending...' };
    members.unshift(m);
    localStorage.setItem('isl_tracker', JSON.stringify(members));
    toggleAddModal(false);
    renderTable();
}

function deleteMember(id) {
    members = members.filter(m => m.id !== id);
    localStorage.setItem('isl_tracker', JSON.stringify(members));
    renderTable();
}

// Boot
window.onload = () => {
    document.getElementById('inputClientId').value = settings.clientId;
    document.getElementById('inputSheetId').value = settings.sheetId;
    renderTable();
};
